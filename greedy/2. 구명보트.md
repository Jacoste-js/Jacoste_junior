function solution(people, limit) {
    let answer = 0;
    people.sort((a, b) => a - b);
    while (people.length) {
        answer++;
        let w1 = people.pop();
        if (people.length && w1 + people[0] <= limit)
            people.shift();
    }
    return answer;
}
